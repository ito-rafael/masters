# back to browser
    --> Workloads
    --> Add
    --> Add with YAML
-----
apiVersion: apps/v1
kind: Deployment
metadata:
  name: laravel-crud-app
spec:
  replicas: 1
  selector:
    matchLabels:
      app: laravel-crud-app
  template:
    metadata:
      labels:
        app: laravel-crud-app
    spec:
      containers:
        - name: com320-app
          image: com320.azurecr.io/laravel-crud-app
          imagePullPolicy: Always
          ports:
          - containerPort: 80
-----

# back to browser
    --> Services and ingresses

apiVersion: v1

  name: lb-com320-app
spec:
  type: LoadBalancer
  selector:
    app: laravel-crud-app
  ports:
    - protocol: TCP
      port: 8080
      targetPort: 80
-----
